function criarSala() {
    const salaId = Math.random().toString(36).substring(2, 8);
    firebase.database().ref('salas/' + salaId).set({
        cartas: [],
        jogador1Pronto: false,
        jogador2Pronto: false
    }).then(() => {
        window.location.href = 'jogo.html?sala=' + salaId;
    });
}

function entrarSalaPrompt() {

    }
}
